Transfer your data and codes to Sherlock cluster

rsync -a --stats ~/Desktop/SLcodes steeve@sherlock.stanford.edu:/home/steeve/proj/
rsync -a --stats ~/Desktop/mgl steeve@sherlock.stanford.edu:/home/steeve/proj/
rsync -a --stats ~/Desktop/mrTools steeve@sherlock.stanford.edu:/home/steeve/
proj/

Then write a "slWJMMAPReadout01.matlab" file and save it in /home/steeve/ on Sherlock.

#!/bin/bash
#BATCH --job-name=slWJMMAPReadout01
#SBATCH --output=slWJMMAPReadout01.out
#SBATCH --error=slWJMMAPReadout01.err
#SBATCH --time=167:00:00
#SBATCH --qos=long
#SBATCH --nodes=1
#SBATCH --ntasks-per-node=16
#SBATCH --export=All
#SBATCH --mem-per-cpu 4000
#SBATCH --mem 64000
#SBATCH --exclusive
#SBATCH --mail-type=ALL
##my commands
##load matlab
ml load matlab
##add code library path
matlab -nojvm -r "addpath(genpath('~/proj')); slFitWJM({'sub01'},[3 0.7 2.7 8 33 1e-2 32],'pscaling',ones(1,7),'MAPReadout','experiment','vonMisesPrior','fminsearch','sherlock')" outputWJMMAPReadout01.log

Submit job

sbatch slWJMMAPReadout01.matlab
